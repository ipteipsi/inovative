entra(Populaçao individuos[])

	dimPop := tamanho(individuos[])
	dimGene := tamanho(individuos[].genes[])

	para i := 0 ate i = dimPop 
		alelo1 := aleatorio(0<=rand<dimGene)
		fazer
			alelo2 := aleatorio(0<=rand<dimGene)
		enquanto(alelo1 = alelo2)

		individuos[i] := troca(individuos[i], alelo1, alelo2)
		i := i+1

sai(Populaçao individuos[])



troca(Individuo pessoa, alelo1, alelo2)

	aleloTmp := pessoa.genes[alelo1]
	pessoa.genes[alelo1] := pessoa.genes[alelo2]
	pessoa.genes[alelo2] := aleloTmp
		
retorna(Individuo pessoa)
