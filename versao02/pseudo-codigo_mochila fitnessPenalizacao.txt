fitnessPenal(Individuo genes[])
	pesoMax
	peso[]
	valor[]
	para i := 0 até i = tamanho(peso[])-1
		se genes[i] = 1 entao
			val := val+valor[i]
			pes := pes+peso[i]
		i := i+1
	se pes <= pesoMax então
		fitness := val
	senão
		fitness := val-(pes-pesoMax)
	se fitness < 0 entao
		fitness := 0
	retorna fitness
